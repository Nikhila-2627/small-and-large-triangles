#include <stdio.h>
#include <stdlib.h>
#include <math.h>

struct triangle
{
	int a;
	int b;
	int c;
};

typedef struct triangle triangle;

    double getAreaSquared(struct triangle t) {
    double s = (t.a + t.b + t.c) / 2.0;
    return s * (s - t.a) * (s - t.b) * (s - t.c);
    }

// Comparator function for qsort
int compare(const void *x, const void *y) {
    struct triangle *t1 = (struct triangle *)x;
    struct triangle *t2 = (struct triangle *)y;

    double area1 = getAreaSquared(*t1);
    double area2 = getAreaSquared(*t2);

    if (area1 > area2)
        return 1;
    else if (area1 < area2)
        return -1;
    else
        return 0;
}

// Required function in HackerRank
void sort_by_area(struct triangle* tr, int n) {
    qsort(tr, n, sizeof(struct triangle), compare);
}


int main()
{
	int n;
	scanf("%d", &n);
	triangle *tr = malloc(n * sizeof(triangle));
	for (int i = 0; i < n; i++) {
		scanf("%d%d%d", &tr[i].a, &tr[i].b, &tr[i].c);
	}
	sort_by_area(tr, n);
	for (int i = 0; i < n; i++) {
		printf("%d %d %d\n", tr[i].a, tr[i].b, tr[i].c);
	}
	return 0;
}
